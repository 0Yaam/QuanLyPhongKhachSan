/* ========== 4) VIEW TỔNG HỢP HÓA ĐƠN (tính nhanh tổng phải thu) ========== */
CREATE VIEW [dbo].[vw_HoaDon_Tong] AS
SELECT
  h.MaHD,
  h.MaDat,
  h.NgayLap,
  h.LoaiHoaDon,
  SUM(ISNULL(ct.ThanhTien,0)) AS TongChiTiet,
  ISNULL(h.TienCoc,0)         AS TienCoc,
  ISNULL(h.TienHoanTra,0)     AS TienHoanTra,
  ISNULL(h.TienBoiThuong,0)   AS TienBoiThuong,
  CAST(SUM(ISNULL(ct.ThanhTien,0))
       - ISNULL(h.TienCoc,0)
       - ISNULL(h.TienHoanTra,0)
       + ISNULL(h.TienBoiThuong,0) AS DECIMAL(18,2)) AS TongPhaiThu
FROM dbo.HoaDon h
LEFT JOIN dbo.ChiTietHoaDon ct ON ct.MaHD = h.MaHD
GROUP BY h.MaHD, h.MaDat, h.NgayLap, h.LoaiHoaDon, h.TienCoc, h.TienHoanTra, h.TienBoiThuong;
GO
